CREATE DATABASE IF NOT EXISTS bdchu;

CREATE TABLE Patients
(
IdPatient int NOT NULL PRIMARY KEY,
Nom varchar(255) NOT NULL,
Prenom varchar(255) NOT NULL,
DateNaissance date NOT NULL,
AdressPostale varchar(255) NOT NULL,
NumSecu int NOT NULL UNIQUE
);

CREATE TABLE Services
(
IdServices int NOT NULL PRIMARY KEY,
NomServices varchar(255) NOT NULL
);

CREATE TABLE Consultations
(
IdConsul int NOT NULL PRIMARY KEY,
fk_IdPatient int NOT NULL,
fk_IdServices int NOT NULL,
DateConsul date NOT NULL,
ButConsul varchar(255) NOT NULL
);
