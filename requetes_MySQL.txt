Affichage fiche patient

SELECT patients.Nom, patients.Prenom, patients.DateNaissance, patients.NumSecu, patients.AdressPostale, consultations.DateConsul, consultations.ButConsul, services.NomService 
FROM patients 
INNER JOIN consultations 
ON consultations.fk_IdPatient=patients.IdPatient 
INNER JOIN services 
ON consultations.fk_IdService=services.IdService 
WHERE Nom='Nom'
